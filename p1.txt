class RedRobin {

  RedRobin() {
    number a, b, c, d;
    if (a >= b) {
      if (b <= a) {
        a = b + c;
      }
      else {
        c = d + a;
      }
      d = d + 1;
    }
    else {
        c = b * a;
    }
    a = b + c;

    for c in (2 _ 5) step 1 {
      a = a + 10;
    }
    
  }

}
